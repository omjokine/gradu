\begin{frame}
    \frametitle{HY-Beamer}
    \framesubtitle{Käyttö}

    \begin{itemize}
      \item Käyttöesimerkkinä tämän kalvosetin lähdekoodi

        \bigskip
      \item Hyödyllisiä \texttt{HY}-teeman optioita
        \begin{itemize}
          \item \texttt{sinetti} -- opintokirjaroviotunnuksen sijasta
            yliopiston sinetti
          \item \texttt{ml} -- yliopiston sinisen perusvärityksen
            sijasta ML-tiedekunnan oranssi
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{HY-Beamer}
    \framesubtitle{Kirjaimet}

    \begin{itemize}
        \bigskip
      \item Beamer ei ota kantaa kirjaimiin

        \bigskip
      \item Tässä otettu perus-PS-kirjainperheet \textsf{Helvetica},
        \textrm{Times} ja \texttt{Courier} (myös matematiikkaan:
        $\sum_{i=1}^{64}\mathsf{log}(i+i^2)$)

        \bigskip
      \item Yliopiston käytäntö: kalvojen teksti pääteviivatonta
        (virallisesti Frutiger, suositeltuja korvikkeita Arial ja
        Helvetica)
    \end{itemize}
\end{frame}
